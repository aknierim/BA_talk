\begin{frame}{Evaluation Metrics}

            \tpr &= \frac{\tp}{\tp + \fn} \\
            \tnr &= \frac{\tn}{\tn + \fp} \\
            \fnr &= \frac{\fn}{\fn + \tp} \\
            \acc &= \frac{\tp + \tn}{\tp + \fp + \tn + \fn} \\
            \ba &= \frac{\tpr + \tnr}{2}
