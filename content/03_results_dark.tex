\subsection{Metrics}%
\label{sub:Metrics}

\begin{frame}{Evaluation Metrics}
    \begin{minipage}{0.48\textwidth}
        \begingroup
        \setlength{\jot}{10pt}
        \begin{align*}
            \tpr &= \frac{\tp}{\tp + \fn} \\
            \tnr &= \frac{\tn}{\tn + \fp} \\
            \fnr &= \frac{\fn}{\fn + \tp} \\
            \acc &= \frac{\tp + \tn}{\tp + \fp + \tn + \fn} \\
            \ba &= \frac{\tpr + \tnr}{2}
        \end{align*}
        \endgroup
    \end{minipage}
    \begin{minipage}{0.48\textwidth}
        \begin{table}
            \centering
            \begin{tblr}{

            & & positive & negative \\

            & neg. & False Positive ($\fp$) & True Negative ($\tn$) \\
            \end{tblr}
        \end{table}
